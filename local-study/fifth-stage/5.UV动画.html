// 如果是U动画，需要设置wrapS重复
texture.wrapS = THREE.RepeatWrapping;

// 如果是V动画，需要设置warpT重复
texture.wrapT = THREE.RepeatWrapping;

// 渲染函数
function render() {
    texture.offset.x += 0.001; // U动画
    renderer.render(scene, camera); //执行渲染操作
    requestAnimationFrame(render); //请求再次执行渲染函数render，渲染下一帧
}
render();
